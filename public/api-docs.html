              POST
              /api/jobs
              — Criar nova transcrição
            

          

          

            

              Envie como multipart/form-data.
              Resposta retorna id imediatamente; o processamento ocorre em background.
            

            

              
Campos:

              
    
                
  • audio — arquivo (obrigatório)
    • 
                
  • title — título (opcional)
    • 
                
  • mode — transcribe ou align
    • 
                
  • language — pt, hr, en, etc.
    • 
                
  • device — cpu ou cuda
    • 
                
  • model_size — tiny, base, small, medium, large-v2, large-v3
    • 
                
  • compute_type — opcional (auto-corrigido)
    • 
                
  • reference_text — obrigatório no modo align
    • 
              

            

            

              
curl

              
curl -X POST http://localhost:9000/api/jobs \
  -F "audio=@meu-audio.mp3" \
  -F "title=Episodio 01" \
  -F "mode=transcribe" \
  -F "language=pt" \
  -F "device=cpu" \
  -F "model_size=small"

              DELETE
              /api/jobs/:id
              — Apagar job e seus arquivos
            

          

          

            

              Remove o job do banco e apaga os arquivos físicos (áudio + saídas SRT/VTT/TXT/JSON).
              Operação irreversível.
            

          

        

      


      
      

        

          

            

              GET
              /api/jobs/:id/audio
              — Streaming do áudio original
            

          

          

            

              Suporta Range requests para HTML5 audio. Use direto em <audio src="...">
              ou em players customizados.
            

          

        

      


      
      

        

          

            

              PUT
              /api/jobs/:id/segments
              — Salvar edições manuais nos segmentos
            

          

          

            

              Substitui todos os segmentos pelo array fornecido. Os arquivos SRT/VTT/TXT/JSON são regravados automaticamente.
            

            

              
Body (application/json)

              
{
  "segments": [
    { "start": 0.27, "end": 5.0,  "text": "Primeira fala...", "words": [] },
    { "start": 5.0,  "end": 10.0, "text": "Segunda fala...",  "words": [] }
  ]
}

            

          

        

      


      
      

        

          

            

              POST
              /api/jobs/:id/regroup
              — Reagrupar palavras em legendas
            

          

          

            

              Redistribui as palavras (mantendo timestamps) em segmentos seguindo regras configuráveis.
              Útil para ajustar legendas ao padrão Netflix, TikTok, etc.
            

            

              
Body (application/json)

              
{
  "max_chars_per_line": 42,
  "max_lines_per_segment": 2,
  "max_duration_seconds": 7,
  "min_duration_seconds": 1.0,
  "break_on_punctuation": true,
  "merge_short_segments": true,
  "min_words_per_segment": 2
}

            

            

              Presets prontos:
              Netflix (42×2), TikTok (25×1)
            

          

        

      


      
      

        

          

            

              POST
              /api/jobs/:id/postprocess
              — Pós-processamento do texto
            

          

          

            

              Aplica regras de limpeza no texto (capitalização, pontuação, disfluências).
            

            

              
Body (application/json)

              
{
  "language": "pt",
  "capitalize_first_word": true,
  "capitalize_after_period": true,
  "fix_punctuation_spacing": true,
  "collapse_spaces": true,
  "remove_filler_words": false,
  "add_period_at_end": false
}

            

          

        

      


      
      

        

          

            

              GET
              /api/jobs/:id/download/:format
              — Baixar legendas geradas
            

          

          

            

              :format aceita: srt, vtt, txt, json.
              Retorna o arquivo com header de download apropriado.
            

            

              /download/srt
              /download/vtt
              /download/txt
              /download/json

Pipelines prontos para automatizar

          

          


            
            

              

                ▸ Bash — pipeline completo (transcrever e baixar SRT)
              

              
BASE=http://localhost:9000

# 1. Cria job e captura o ID
JOB_ID=$(curl -s -X POST $BASE/api/jobs \
  -F "audio=@audio.mp3" \
  -F "language=pt" \
  -F "mode=transcribe" \
  -F "model_size=small" \
  -F "device=cpu" | jq -r '.id')

echo "Job criado: $JOB_ID"

# 2. Aguarda processar (polling)
while true; do
  STATUS=$(curl -s $BASE/api/jobs/$JOB_ID | jq -r '.status')
  echo "Status: $STATUS"
  [ "$STATUS" = "done" ] && break
  [ "$STATUS" = "error" ] && exit 1
  sleep 5
done

# 3. Aplica reagrupamento estilo Netflix
curl -s -X POST $BASE/api/jobs/$JOB_ID/regroup \
  -H "Content-Type: application/json" \
  -d '{"max_chars_per_line":42,"max_lines_per_segment":2}'

# 4. Baixa SRT final
curl -s -o legenda.srt $BASE/api/jobs/$JOB_ID/download/srt
echo "Legenda salva em legenda.srt"

            


            
            

              

                ▸ Node.js — automação com axios
              

              
const axios = require('axios');
const fs = require('node:fs');
const FormData = require('form-data');

const BASE = 'http://localhost:9000';

async function transcribe(audioPath) {
  const fd = new FormData();
  fd.append('audio', fs.createReadStream(audioPath));
  fd.append('language', 'hr');
  fd.append('mode', 'transcribe');
  fd.append('model_size', 'small');
  fd.append('device', 'cpu');

  const { data } = await axios.post(`${BASE}/api/jobs`, fd, {
    headers: fd.getHeaders(),
  });
  const id = data.id;
  console.log('Job:', id);

  // polling
  while (true) {
    const job = (await axios.get(`${BASE}/api/jobs/${id}`)).data;
    console.log('Status:', job.status, '-', job.progress_message || '');
    if (job.status === 'done') {
      const srt = await axios.get(`${BASE}/api/jobs/${id}/download/srt`,
        { responseType: 'arraybuffer' });
      fs.writeFileSync('output.srt', srt.data);
      return id;
    }
    if (job.status === 'error') throw new Error(job.error_message);
    await new Promise(r => setTimeout(r, 3000));
  }
}

transcribe('./audio.mp3').then(console.log);

            


            
            

              

                ▸ Python — modo "alinhar" (TTS)
              

              
import requests, time

BASE = 'http://localhost:9000'

with open('roteiro.txt', encoding='utf-8') as f:
    roteiro = f.read()

# Cria job em modo align (sincroniza texto exato com audio TTS)
files = {'audio': open('tts-output.mp3', 'rb')}
data = {
    'mode': 'align',
    'language': 'hr',
    'reference_text': roteiro,
    'device': 'cpu',
}
r = requests.post(f'{BASE}/api/jobs', files=files, data=data)
job_id = r.json()['id']
print(f'Job: {job_id}')

# Polling
while True:
    job = requests.get(f'{BASE}/api/jobs/{job_id}').json()
    print(f"Status: {job['status']} - {job.get('progress_message', '')}")
    if job['status'] == 'done':
        srt = requests.get(f'{BASE}/api/jobs/{job_id}/download/srt')
        open('legenda.srt', 'wb').write(srt.content)
        break
    if job['status'] == 'error':
        raise Exception(job['error_message'])
    time.sleep(3)
